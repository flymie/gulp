xwPop.msg简便用法
1、info xw.info(text,callBack)
2、success xw.success(text,callBack)
3、error xw.error(text,callBack)
4、warn xw.warn(text,callBack)

xwpop.msg完整用法
xwPop.msg({
	type:'info',   //success info warning error
	callBack:'', //回调函数
	showClose:false, //是否显示关闭按钮
	text:'提示的文字内容', //内容
	duration:3000, //显示的时间
	fadeTime:500 //渐隐 渐入的时间
})

xwPop.modal 简便用法
xw.alert(txt,callBack)
xw.confirm(txt,callBack,returnCallBack)

xwPop.modal 详细用法
xwPop.modal({
	title:'提示',
	type:'alert', //alert confirm
	callBack:'', //确定回调
	returnCallBack:'', //取消回调 confirm下才有
	text: '提示的文字内容',
	btn:{ //按钮文字
		yes: '确定',
		no: '取消'
	},
	autoClose:true, //点击确定取消执行function后 是否关闭当前弹窗
	fadeTime:500, //渐隐 渐显的时间
	width:'100%', //宽度
	mask:true,//是否有蒙层
	location:"btm", //默认在最底层 两种可选‘btm、mid’
	hasClose:true,  //默认有关闭按钮
})
